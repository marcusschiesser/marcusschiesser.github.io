    <![CDATA[
        import mx.core.UIComponent;
        import mx.controls.Label;
        import com.google.maps.MapMouseEvent;
        import com.google.maps.overlays.Marker;
        import com.google.maps.controls.ZoomControl;
        import com.google.maps.LatLng;
        import com.google.maps.MapType;
        import com.google.maps.MapEvent;
        
        private var HOME:LatLng = new LatLng(
                    49.003455, 8.407934
                    );
        
        private function ready(event:MapEvent):void {
            flexMap.map.setMapType(MapType.NORMAL_MAP_TYPE);

            flexMap.map.setZoom(14);
            flexMap.map.setCenter(HOME);
                    
            flexMap.map.addControl(new ZoomControl());
            
            flexMap.map.addOverlay(createMarker());
        }
        
        private function createMarker():Marker {
            var marker:Marker = new Marker(HOME);
            marker.addEventListener(MapMouseEvent.CLICK,
                function(event:MapMouseEvent):void {
                    flexMap.map.openFlexInfoWindows(HOME, [(address as ClassFactory).newInstance(),
                    (pic as ClassFactory).newInstance()])
                }
                );
            return marker;
        }
